-- How old is someone by surname (prefix match — Hill matches Hill, not Chhill)
SELECT CustomerId, Surname, Age
FROM Customers
WHERE Surname LIKE 'Hargrave%'
LIMIT 5;

-- Where is someone from (Geography)
SELECT CustomerId, Surname, Geography
FROM Customers
WHERE Surname LIKE 'Hargrave%'
LIMIT 1;

-- Credit score by surname
SELECT CustomerId, Surname, CreditScore
FROM Customers
WHERE Surname LIKE 'Hill%'
LIMIT 5;

-- Lookup by customer id
SELECT CustomerId, Surname, Age, CreditScore
FROM Customers
WHERE CustomerId = 123456
LIMIT 1;

-- Average credit score by geography
SELECT Geography, AVG(CreditScore) AS avg_credit_score
FROM Customers
GROUP BY Geography;

-- Compare credit scores between two customers
SELECT CustomerId, Surname, CreditScore
FROM Customers
WHERE Surname LIKE 'Hargrave%' OR Surname LIKE 'Hill%'
ORDER BY CreditScore DESC
LIMIT 10;

-- Compare ages between two customers
SELECT CustomerId, Surname, Age
FROM Customers
WHERE Surname LIKE 'Hargrave%' OR Surname LIKE 'Hill%'
ORDER BY Age DESC
LIMIT 10;

-- Who has the higher balance
SELECT CustomerId, Surname, Balance
FROM Customers
WHERE Surname LIKE 'Hargrave%' OR Surname LIKE 'Hill%'
ORDER BY Balance DESC
LIMIT 10;

-- Compound "and" question: total customers AND male count (answer both parts)
SELECT
  COUNT(*) AS total_customers,
  SUM(CASE WHEN Gender = 'Male' THEN 1 ELSE 0 END) AS male_count
FROM Customers;

-- Compound "and" question: total customers AND average age
SELECT
  COUNT(*) AS total_customers,
  AVG(Age) AS average_age
FROM Customers;
